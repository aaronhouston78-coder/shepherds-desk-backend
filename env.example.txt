# ── Anthropic ─────────────────────────────────────────────────────────────────

# ── Trial anti-abuse ──────────────────────────────────────────────────────────

# ── Support / upgrade contact ──────────────────────────────────────────────────

# ── Email verification (Resend) ───────────────────────────────────────────────
# Required for production email verification.
# Without this key, the server runs in dev mode:
#   — verification links are printed to the console instead of being emailed
#   — local testing works fully without any provider setup
#
# TO ACTIVATE FOR PRODUCTION:
#   1. Sign up at https://resend.com (free tier: 3,000 emails/month)
#   2. Add and verify your sender domain in the Resend dashboard
#   3. Generate an API key and paste it below
#   4. Set EMAIL_FROM to a verified address on that domain
#
RESEND_API_KEY=
EMAIL_FROM=Shepherd's Desk <noreply@yourdomain.com>

# ── Environment ───────────────────────────────────────────────────────────────
# Set to "production" in deployment to activate:
#   — strict CORS (null origin rejected)
#   — suppressed internal error details in responses
#   — reduced startup logging
NODE_ENV=development

# ── Stripe ────────────────────────────────────────────────────────────────────
# 1. Create a Stripe account at https://stripe.com
# 2. In the dashboard, create three Products (Starter / Growth / Church Team)
#    each with a recurring monthly price.
# 3. Copy the secret key and each Price ID below.
# 4. Use test keys (sk_test_...) during development.
# 5. For the webhook:
#    - Install the Stripe CLI: https://stripe.com/docs/stripe-cli
#    - Run: stripe listen --forward-to localhost:3001/api/billing/webhook
#    - Copy the webhook signing secret it prints
# 6. In production, register the webhook URL in the Stripe dashboard under
#    Developers → Webhooks → Add endpoint
#    URL: https://your-backend-domain.com/api/billing/webhook
#    Events to listen for:
#      checkout.session.completed
#      customer.subscription.created
#      customer.subscription.updated
#      customer.subscription.deleted
#      invoice.payment_failed

STRIPE_SECRET_KEY=sk_test_your_secret_key_here
STRIPE_WEBHOOK_SECRET=whsec_your_webhook_signing_secret

# Price IDs — copy from Stripe dashboard after creating each product
STRIPE_PRICE_STARTER=price_starter_id_here
STRIPE_PRICE_GROWTH=price_growth_id_here
STRIPE_PRICE_TEAM=price_team_id_here
